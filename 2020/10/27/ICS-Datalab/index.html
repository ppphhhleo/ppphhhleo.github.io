/* mZ * byteSwap - swaps the nth byte and the mth byte
*  Examples: byteSwap(0x12345678, 1, 3) = 0x56341278
*            byteSwap(0xDEADBEEF, 0, 2) = 0xDEEFBEAD
*  You may assume that 0 <= n <= 3, 0 <= m <= 3
*  Legal ops: ! ~ & ^ | + << >>
*  Max ops: 25
*  Rating: 2
*/
int byteSwap(int x, int n, int m) {
/*
* first rotate n and m right for 3, 
* to find the exact bytes needed to swap
* 
* then we need to mask 1 protecting the irrelevant bytes
* mask 0 to nth and mth bytes
* 
* finally use OR to place to the correct position
*/
int n_shift = n << 3;
int m_shift = m << 3;
int mid = ((x >> m_shift) ^ (x >> n_shift)) & 0xff;
return x ^ ((mid << m_shift) | (mid << n_shift));

/*
version 1:
int shift_n = n << 3;   // * 8
int shift_m = m << 3;   // * 8
int nn = x >> shift_n ;  // to the right
int mm = x >> shift_m;  // rotate to the right
int byte_n = (nn & 0xff) << shift_m; // move to the destination 
int byte_m = (mm & 0xff) << shift_n;
int mask_m = 0xff << shift_m;
int mask_n = 0xff << shift_n;
int mask = ~ (mask_m | mask_n);  // mask swapbytes : 0
int final = mask & x;  // mask on the bytes
int changebyte = byte_n | byte_m; // swap result
return (changebyte | final);
*/
}


/* 
* rotateRight - Rotate x to the right by n
*   Can assume that 0 <= n <= 31
*   Examples: rotateRight(0x87654321,4) = 0x18765432
*   Legal ops: ~ & ^ | + << >> !
*   Max ops: 25
*   Rating: 3 
*/


int rotateRight(int x, int n) {
/*  
version 1:
int right = x >> n;
int i = ~n + 1;
int k = ~ (((0x1 << 31) >> 31) << n);
int c = k & x;  // save the right
int kk = k << (32 + i); // left mask
int kkk = ~kk;
int change = c << (32 + i);
int now = kkk & right; 
int final = now | change;
return final;
*/
/*
 *
 * rotate right first
 * mask lower n bits to 1, 
 * mask higher 32-n bits to 0
 * 
 * rotate left to get the bits that would have lost when rotating right
 * 
 * OR operation, solved
 * 
 */
int right = x >> n;
int neg_n = ~n + 1;
int mask = ~ (~0 << (32 + neg_n));  // higher 32-n bits:0
int left = x << (32 + neg_n);
return left | (mask & right );
}
/* 



* logicalNeg - implement the ! operator, using all of 
*              the legal operators except !
*   Examples: logicalNeg(3) = 0, logicalNeg(0) = 1
*   Legal ops: ~ & ^ | + << >>
*   Max ops: 12
*   Rating: 4 
*/

int logicalNeg(int x) {
/*
*
* unsign the value 
* if x is zero, ~x + 1 = 000...000, the flag sign does not change
* non zero otherwise
* 
* so if x is nonzero, then afterthe highest bit of z is 1
* then get the MSB,  
* add 1
* transform non zero 1 to overflow 0
* transform zero 0 to 1
* 
* done
* 
*/
int y = ~x + 1; 
int z = y | x;  
y = z >> 31;
x = y + 0x1;
return x;
}

int sign(int x) {
/*
*
* if negative, sign is 1
* if positive or all-zero, sign is 0
* 
* if non-zero, !! operation return 1,
* all-zero otherwise
* 
* make a comparism

return sign | compare;
}

/* 
* isGreater - if x > y  then return 1, else return 0 
*   Example: isGreater(4,5) = 0, isGreater(5,4) = 1
*   Legal ops: ! ~ & ^ | + << >>
*   Max ops: 24
*   Rating: 3

int isGreater(int x, int y) {
/*
version 1
int minus = x + ~y + 1;
int minus_sign = (minus >> 31) & 0x1;
int y_sign = y >> 31;
int x_sign = x >> 31;
int x_y = (y_sign ^ x_sign) & 0x1;  // same:0, not same:1
// int x_sub = (sub_sign ^ x_sign) & 0x1; // same:0, not same:1
int same = !minus;  // same: 1, not same: 0
int case1 = (!minus_sign) & (!x_y) & !same; // ans: positive, and, xy same sign, and xy not same
int case2 = x_y & (!x_sign); // xy not same , but x positive:0
return case1 | case2;
*/

/*
* 
* x is greater than y, meanning x - y > 0
* there are two cases when the x - y > 0 :
*    x and y have the same flag sign, and the minus result's flag is positive
 *    x and y have different flag signs, but x's flag sign is positive
* in order not to consider x and y are equal (less ops),
* we consider the opposite cases, then return !
* done
 */
int sign_y = y >> 31;
int sign_x = x >> 31;
int sign_minus = (x + ~y) >> 31;
int sign_notequal = sign_x & !sign_y;  // different sign 
int sign_equal = (!(sign_x ^ sign_y)) & sign_minus; // same sign
return !(sign_equal | sign_notequal);
/*

int y_sign = !!(y >> 31);   // positive:0, negative:1
int x_y = x_sign ^ y_sign; // same:0, not same:1

/* 
* subOK - Determine if can compute x-y without overflow
*   Example: subOK(0x80000000,0x80000000) = 1,
*            subOK(0x80000000,0x70000000) = 0, 
*   Legal ops: ! ~ & ^ | + << >>
*   Max ops: 20
*   Rating: 3
*/
int subOK(int x, int y) {
/*
* the sub is not OK when 
* answer's sign is differen with x's while x's different with y's 
* then ! 
* done
*/
int sign_xy = (x ^ y) >> 31 ; // x and y's signs: same:0, not same:1
int sign_x = x >> 31;
int ans = x + ~y + 1;
int sign_ans = ans >> 31;
int sign_ansx = sign_ans ^ sign_x; // answer's sign and x's sign: same:0 not same:1
return !(sign_ansx & sign_xy); 
}
/*
* satAdd - adds two numbers but when positive overflow occurs, returns
*  maximum possible value, and when negative overflow occurs,
*  it returns minimum positive value.
*  Examples: satAdd(0x40000000,0x40000000) = 0x7fffffff
*            satAdd(0x80000000,0xffffffff) = 0x80000000
*   Legal ops: ! ~ & ^ | + << >>
*   Max ops: 30
*   Rating: 4
*/

int satAdd(int x, int y) {
/*
 positive overflow occurs:  overflow = 111111..111, (overflow & 0x1f) = 31
 * negative overflow occurs: overflow = 0000...000 , (overflow & 0x1f) = 0
 *
 * overflow occurs when sign of adding answers is not the same with both x and y
 *
 * 0x7fffffff = 0x01111...111
 * 0x80000000 = 0x10000...000
 *
 * positive overflow: sign_ans = 1, ans >> 31 + 1000...0000 = 11111..11
 * negative overflow: sign_ans = 0, ans >> 31 + 1000...0000 = 1000...00
 * no overflow: ans >> 0 + 0 = ans
 * solved
 *
 */

int ans = x + y;
int overflow = ((x ^ ans) & (y ^ ans)) >> 31; // overflow:111111..111
return (ans >> (overflow & 0x1f)) + (overflow << 31);

/* howManyBits - return the minimum number of bits required to represent x in two's complement
*  Examples: howManyBits(12) = 5
*            howManyBits(298) = 10
*            howManyBits(-5) = 4
*            howManyBits(0)  = 1
*            howManyBits(-1) = 1
*            howManyBits(0x80000000) = 32
*  Legal ops: ! ~ & ^ | + << >>
*  Max ops: 90
*  Rating: 4
*/

int howManyBits(int x) {
/*
*
* we need to find the first 1 in two's complement 
* if positive, need another flag sign bit
* if negative, no need for flag
* 
* so if x is negative, x ^ sign = ~x 
* then add up the number of all bits lower than the first 1, and the flag
* 'dichotomy'
* first check the higher 16 bits
* if existing, x at least needs lower 16 bits
* 
* if positive, need another flag sign bit
* if negative, no need for  
* then check higher 8 bits of the higher 16 bits
* if existing, x at least nees lower 16 + 8 = 24 bits
* 
* ......
* 
* if nothing exists but 0, we still +1 as the same with flag sign
*/


int bits16, bits8, bits4, bits2, bits1, bits0;
int sign = x >>31;
// x = (sign & ~x) | (~sign & x);
x = x ^ sign;  // 正则不变，否则按位取反，原本最高位为0的正数，也可以将符号位去掉
bits16 = !!(x >> 16) << 4;
x = x >> bits16;
bits8 = !!(x >> 8) << 3;
x = x >> bits8;
bits4 = !!(x >> 4) << 2;
x = x >> bits4;
bits2 = !!(x >> 2) << 1;
x = x >> bits2;
bits1 = !!(x >> 1);
x = x >> bits1;
bits0 = x;
return bits16 + bits8 + bits4 + bits2 + bits1 + bits0 + 1;  // +1 means flag bit

/*
* another similar version

int sign = x >> 31;
int bits;
x = (sign & ~x) | (~sign & x);
bits = (!!(x >> 16) << 4);
bits = bits + ((!!(x >> (bits + 8))) << 3);
bits = bits + ((!!(x >> (bits + 4))) << 2);
bits = bits + ((!!(x >> (bits + 2))) << 1);
bits = bits + (!!(x >> (bits + 1)));
bits = bits + (!!(x >> bits));
return bits + 1;
*/


}
/*
* ilog2 - return floor(log base 2 of x), where x > 0
*   Example: ilog2(16) = 4
*   Legal ops: ! ~ & ^ | + << >>
*   Max ops: 90
*   Rating: 4
*/
int ilog2(int x) {
/*
* the value of 2^n has the form: 000..00100...00
* we can use 'dichotomy' similar to the 'howmanybits'
* we need to find the highest 1
* then add all bits lower the it, whose sum means the index of the former value
* 
* first check the higher 16 bits whether 1 exists
* if existing, x at least needs lower 16 bits
* 
* if positive, need another flag sign bit
* if negative, no need for  
* then check higher 8 bits of the higher 16 bits
* if existing, x at least nees lower 16 + 8 = 24 bits
* 
* ......
* 
*/
int bits16, bits8, bits4, bits2, bits1;
bits16 = (!!(x >> 16) << 4);
x = x >> bits16;
bits8 = (!!(x >> 8) << 3);
x = x >> bits8;
bits4 = (!!(x >> 4) << 2);
x = x >> bits4;
bits2 = (!!(x >> 2) << 1);
x = x >> bits2;
bits1 = (!!(x >> 1));
return bits16 + bits8 + bits4 + bits2 + bits1;
/*
* another similar version
*
int a = (!!(x >> 16) << 4);
a = a + ((!!(x >> (a + 8))) << 3);
a = a + ((!!(x >> (a + 4))) << 2);
a = a + ((!!(x >> (a + 2))) << 1);
a = a + (!!(x >> (a + 1)));
return a;
*/
}
